CREATE PROCEDURE ADD_PROD_COND_STATUS (
  SSN_ID VARCHAR(20),
  CODE VARCHAR(1),
  DESCRIPTION VARCHAR(20)
) RETURNS (
  ADD_PROD CHAR(1)
) AS  
      
  DECLARE VARIABLE REC_EXIST INTEGER; 
BEGIN
  REC_EXIST = 0;
  ADD_PROD = 'F'; 
  
  /* Check if record exists */
  SELECT 1
  FROM PRODUCT_COND_STATUS
  WHERE SSN_ID = :SSN_ID
  AND CODE = :CODE
  AND DESCRIPTION = :DESCRIPTION  
  INTO :REC_EXIST;  

  /* if record not exists then insert */
  IF (REC_EXIST = 0) THEN
  BEGIN
    ADD_PROD = 'T';
 
    INSERT INTO PRODUCT_COND_STATUS   
         (SSN_ID, CODE, DESCRIPTION)
    VALUES (:SSN_ID, :CODE, :DESCRIPTION);             
  END
END
^
